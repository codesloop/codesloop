LINK_DIRECTORIES(
                  ../../common
                  ../../db
                )

INCLUDE_DIRECTORIES( ../../.. )

LINK_LIBRARIES(
                csl_common
              )

IF(NOT WIN32)

  SET( CMAKE_CXX_FLAGS "-Wconversion -Wcast-qual -Wpointer-arith -Wcast-align -Wold-style-cast" )

  ADD_DEFINITIONS(
                   -Wall
                   -g3
                   -Wwrite-strings
                   -Wcomment
                   -Wshadow
                   -Wredundant-decls
                 )

ELSE(NOT WIN32)

  IF(MSYS)

    SET( CMAKE_CXX_FLAGS "-Wconversion -Wcast-qual -Wpointer-arith -Wcast-align -Wold-style-cast" )

    ADD_DEFINITIONS(
                      -Wall
                      -g3
                      -Wwrite-strings
                      -Wcomment
                      -Wshadow
                      -Wredundant-decls
                   )

  ELSE(MSYS)


  ENDIF(MSYS)

ENDIF(NOT WIN32)

ADD_LIBRARY( csl_db_mysql STATIC
             driver.cc  driver.hh
           )

FILE(GLOB includes "${CMAKE_CURRENT_SOURCE_DIR}/*.h*")
INSTALL( FILES ${includes} DESTINATION include/codesloop/db/mysql )
INSTALL(TARGETS csl_db_mysql
  RUNTIME DESTINATION bin
  LIBRARY DESTINATION lib
  ARCHIVE DESTINATION lib
)
